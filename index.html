async function updateData() { const timeEl = document.getElementById('update-time'); // 我們改用一個專門處理「防鎖定」的代理服務：api.allorigins.win // 這是目前存活率最高、最穩定的代理方式 const targetUrl = "https://api.taiwanlottery.com.tw/TLCAPIWeB/Lottery/BingoBingoResult"; const proxyUrl = "https://api.allorigins.win/get?url=" + encodeURIComponent(targetUrl); try { const response = await fetch(proxyUrl); if (!response.ok) throw new Error("代理伺服器連線失敗"); const rawData = await response.json(); // 解析代理伺服器回傳的 JSON 內容 const res = JSON.parse(rawData.contents); const content = res.content.slice(0, 10); // --- 下面維持原本的 AI 計算邏輯 --- let allNums = []; content.forEach(p => allNums.push(...p.drawNo.map(Number))); let counts = {}; for(let i=1; i<=80; i++) counts[i] = 0; allNums.forEach(n => counts[n]++); let hotNums = Object.keys(counts).sort((a,b) => counts[b] - counts[a]); let predNums = hotNums.slice(0, 9).map(Number).sort((a,b)=>a-b); let superNum = parseInt(hotNums[0]); timeEl.innerText = '最後更新：' + new Date().toLocaleTimeString(); document.getElementById('next-period').innerText = parseInt(content[0].period) + 1; document.getElementById('prediction-balls').innerHTML = predNums.map(n => `
${n}
`).join('') + `
${superNum}
`; document.getElementById('history-table').innerHTML = content.map(p => `${p.period}${p.drawNo.join(', ')}${p.superNo}`).join(''); } catch (e) { timeEl.innerText = '連線超時，請按重新整理'; console.error("連線錯誤:", e); } }
